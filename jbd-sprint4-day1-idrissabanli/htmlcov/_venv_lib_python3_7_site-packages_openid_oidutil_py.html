148def appendArgs(url, args): 

149 """Append query arguments to a HTTP(s) URL. If the URL already has 

150 query arguemtns, these arguments will be added, and the existing 

151 arguments will be preserved. Duplicate arguments will not be 

152 detected or collapsed (both will appear in the output). 

153 

154 @param url: The url to which the arguments will be appended 

155 @type url: str 

156 

157 @param args: The query arguments to add to the URL. If a 

158 dictionary is passed, the items will be sorted before 

159 appending them to the URL. If a sequence of pairs is passed, 

160 the order of the sequence will be preserved. 

161 @type args: A dictionary from string to string, or a sequence of 

162 pairs of strings. 

163 

164 @returns: The URL with the parameters added 

165 @rtype: str 

166 """ 

167 if hasattr(args, 'items'): 

168 args = sorted(args.items()) 

169 else: 

170 args = list(args) 

171 

172 if not isinstance(url, str): 

173 url = str(url, encoding="utf-8") 

174 

175 if not args: 

176 return url 

177 

178 if '?' in url: 

179 sep = '&' 

180 else: 

181 sep = '?' 

182 

183 # Map unicode to UTF-8 if present. Do not make any assumptions 

184 # about the encodings of plain bytes (str). 

185 i = 0 

186 for k, v in args: 

187 if not isinstance(k, bytes): 

188 k = k.encode('utf-8') 

189 

190 if not isinstance(v, bytes): 

191 v = v.encode('utf-8') 

192 

193 args[i] = (k, v) 

194 i += 1 

195 

196 return '%s%s%s' % (url, sep, urlencode(args))
