175 # Entry is an object representing a single entry in the

177 # a hashed value of the meat of each entry on its side in

267 def fix_nls(self, in_, out_): 

268 """Fixes submitted translations by filtering carriage returns and pairing 

269 newlines at the begging and end of the translated string with the original 

270 """ 

271 if 0 == len(in_) or 0 == len(out_): 

272 return out_ 

273 

274 if "\r" in out_ and "\r" not in in_: 

275 out_ = out_.replace("\r", '') 

276 

277 if "\n" == in_[0] and "\n" != out_[0]: 

278 out_ = "\n" + out_ 

279 elif "\n" != in_[0] and "\n" == out_[0]: 

280 out_ = out_.lstrip() 

281 if 0 == len(out_): 

282 pass 

283 elif "\n" == in_[-1] and "\n" != out_[-1]: 

284 out_ = out_ + "\n" 

285 elif "\n" != in_[-1] and "\n" == out_[-1]: 

286 out_ = out_.rstrip() 

287 return out_

289 def post(self, request, *args, **kwargs): 

290 """The only circumstances when we POST is to submit the main form, both 

291 updating translations (if any changed) and advancing to the next page of 

292 messages. 

293 

294 There is no notion of validation of this content; as implemented, unknown 

295 fields are ignored and a generic failure message is shown. 

296 

297 Submitted changes are saved out to the specified .po file on the 

298 filesystem if that file is writable, otherwise the cached version of the 

299 file is updated (so it can be downloaded). Then the user is redirected 

300 to the next page of messages (if there is one; otherwise they're 

301 redirected back to the current page). 

302 """ 

303 # The message text inputs are captured as hashes of their initial 

304 # contents, preceded by "m_". Messages with plurals end with their 

305 # variation number. 

306 single_text_input_regex = re.compile(r'^m_([0-9a-f]+)$') 

307 plural_text_input_regex = re.compile(r'^m_([0-9a-f]+)_([0-9]+)$') 

308 file_change = False 

309 for field_name, new_msgstr in request.POST.items(): 

310 md5hash = None 

311 

312 if plural_text_input_regex.match(field_name): 

313 md5hash, plural_id = plural_text_input_regex.match(field_name).groups() 

314 md5hash = str(md5hash) 

315 # polib parses .po files into unicode strings, but 

316 # doesn't bother to convert plural indexes to int, 

317 # so we need unicode here. 

318 plural_id = six.text_type(plural_id) 

319 

320 # Above no longer true as of Polib 1.0.4 

321 if plural_id and plural_id.isdigit(): 

322 plural_id = int(plural_id) 

323 

324 elif single_text_input_regex.match(field_name): 

325 md5hash = str(single_text_input_regex.match(field_name).groups()[0]) 

326 plural_id = None 

327 

328 if md5hash is not None: # Empty string should be processed! 

329 entry = self.po_file.find(md5hash, 'md5hash') 

330 # If someone did a makemessage, some entries might 

331 # have been removed, so we need to check. 

332 if entry: 

333 old_msgstr = entry.msgstr 

334 if plural_id is not None: # 0 is ok! 

335 entry.msgstr_plural[plural_id] = self.fix_nls( 

336 entry.msgid_plural, new_msgstr 

337 ) 

338 else: 

339 entry.msgstr = self.fix_nls(entry.msgid, new_msgstr) 

340 

341 is_fuzzy = bool(self.request.POST.get('f_%s' % md5hash, False)) 

342 old_fuzzy = 'fuzzy' in entry.flags 

343 

344 if old_fuzzy and not is_fuzzy: 

345 entry.flags.remove('fuzzy') 

346 elif not old_fuzzy and is_fuzzy: 

347 entry.flags.append('fuzzy') 

348 

349 file_change = True 

350 

351 if old_msgstr != new_msgstr or old_fuzzy != is_fuzzy: 

352 entry_changed.send( 

353 sender=entry, 

354 user=request.user, 

355 old_msgstr=old_msgstr, 

356 old_fuzzy=old_fuzzy, 

357 pofile=self.po_file_path, 

358 language_code=self.language_id, 

359 ) 

360 else: 

361 messages.error( 

362 self.request, 

363 _( 

364 "Some items in your last translation block couldn't " 

365 "be saved: this usually happens when the catalog file " 

366 "changes on disk after you last loaded it." 

367 ), 

368 ) 

369 

370 if file_change and self.po_file_is_writable: 

371 try: 

372 self.po_file.metadata['Last-Translator'] = unicodedata.normalize( 

373 'NFKD', 

374 u"%s %s <%s>" 

375 % ( 

376 getattr(self.request.user, 'first_name', 'Anonymous'), 

377 getattr(self.request.user, 'last_name', 'User'), 

378 getattr(self.request.user, 'email', 'anonymous@user.tld'), 

379 ), 

380 ).encode('ascii', 'ignore')

382 get_rosetta_version() 

383 ) 

384 self.po_file.metadata['PO-Revision-Date'] = timestamp_with_timezone() 

385 except UnicodeDecodeError: 

386 pass 

387 

388 try: 

389 self.po_file.save() 

390 po_filepath, ext = os.path.splitext(self.po_file_path) 

391 

392 if rosetta_settings.AUTO_COMPILE: 

393 self.po_file.save_as_mofile(po_filepath + '.mo') 

394 

395 post_save.send( 

396 sender=None, language_code=self.language_id, request=self.request 

397 ) 

398 # Try auto-reloading via the WSGI daemon mode reload mechanism 

399 should_try_wsgi_reload = ( 

400 rosetta_settings.WSGI_AUTO_RELOAD 

401 and 'mod_wsgi.process_group' in self.request.environ 

402 and self.request.environ.get('mod_wsgi.process_group', None) 

403 and 'SCRIPT_FILENAME' in self.request.environ 

404 and int(self.request.environ.get('mod_wsgi.script_reloading', 0))

406 if should_try_wsgi_reload: 

407 try: 

408 os.utime(self.request.environ.get('SCRIPT_FILENAME'), None)

412 if rosetta_settings.UWSGI_AUTO_RELOAD:

601 """Validate/return msg_filter from request (e.g. 'fuzzy', 'untranslated'), 

602 or a default. 

603 

604 If a query is also specified in the request, then return None. 

605 """ 

606 if self.query: 

607 msg_filter = None 

608 else: 

609 msg_filter = self._request_request('msg_filter', 'all') 

610 available_msg_filters = {'untranslated', 'translated', 'fuzzy', 'all'} 

611 if msg_filter not in available_msg_filters: 

612 msg_filter = 'all' 

613 return msg_filter 

614 

615 @cached_property 

616 def query(self): 

617 """Strip and return the query (for searching the catalog) from the 

618 request, or None. 

619 """ 

620 return self._request_request('query', '').strip() or None 

621 

622 def get_entries(self): 

623 """Return a list of the entries (messages) that would be part of the 

624 current "view"; that is, all of the ones from this .po file matching the 

625 current query or msg_filter. 

626 """ 

627 if self.query: 

628 # Scenario #1: terms matching a search query 

629 rx = re.compile(re.escape(self.query), re.IGNORECASE) 

630 

631 def concat_entry(e): 

632 return ( 

633 six.text_type(e.msgstr) 

634 + six.text_type(e.msgid) 

635 + six.text_type(e.msgctxt) 

636 + six.text_type(e.comment) 

637 + u''.join([o[0] for o in e.occurrences]) 

638 + six.text_type(e.msgid_plural) 

639 + u''.join(e.msgstr_plural.values()) 

640 ) 

641 

642 entries = [ 

643 e_ 

644 for e_ in self.po_file 

645 if not e_.obsolete and rx.search(concat_entry(e_)) 

646 ] 

647 else: 

648 # Scenario #2: filtered list of messages 

649 if self.msg_filter == 'untranslated': 

650 entries = self.po_file.untranslated_entries() 

651 elif self.msg_filter == 'translated': 

652 entries = self.po_file.translated_entries() 

653 elif self.msg_filter == 'fuzzy': 

654 entries = [e_ for e_ in self.po_file.fuzzy_entries() if not e_.obsolete] 

655 else: 

656 # ("all") 

657 entries = [e_ for e_ in self.po_file if not e_.obsolete] 

658 return entries

697 def translate(text, from_language, to_language, subscription_key): 

698 """ 

699 This method does the heavy lifting of connecting to the translator API and fetching a response 

700 :param text: The source text to be translated 

701 :param from_language: The language of the source text 

702 :param to_language: The target language to translate the text into 

703 :param subscription_key: An API key that grants you access to the Azure translation service 

704 :return: Returns the response from the AZURE service as a python object. For more information about the 

705 response, please visit 

706 https://docs.microsoft.com/en-us/azure/cognitive-services/translator/reference/v3-0-translate?tabs=curl 

707 """ 

708 

709 AZURE_TRANSLATOR_HOST = 'https://api.cognitive.microsofttranslator.com' 

710 AZURE_TRANSLATOR_PATH = '/translate?api-version=3.0' 

711 

712 headers = { 

713 'Ocp-Apim-Subscription-Key': subscription_key, 

714 'Content-type': 'application/json', 

715 'X-ClientTraceId': str(uuid.uuid4()), 

716 } 

717 

718 url_parameters = {"from": from_language, "to": to_language} 

719 

720 request_data = [{"text": text}] 

721 

722 api_hostname = AZURE_TRANSLATOR_HOST + AZURE_TRANSLATOR_PATH 

723 r = requests.post( 

724 api_hostname, 

725 headers=headers, 

726 params=url_parameters, 

727 data=json.dumps(request_data), 

728 ) 

729 return json.loads(r.text) 

730 

731 language_from = request.GET.get('from', None) 

732 language_to = request.GET.get('to', None) 

733 text = request.GET.get('text', None) 

734 

735 if language_from == language_to: 

736 data = {'success': True, 'translation': text} 

737 else: 

738 # run the translation: 

739 AZURE_CLIENT_SECRET = getattr(settings, 'AZURE_CLIENT_SECRET', None) 

740 

741 try: 

742 api_response = translate( 

743 text, language_from, language_to, AZURE_CLIENT_SECRET 

744 ) 

745 

746 # result will be a dict if there is an error, e.g. 

747 # { 

748 # "success": false, 

749 # "error": "Microsoft Translation API error: Error code 401000, 

750 # The request is not authorized because credentials are missing or invalid." 

751 # } 

752 if isinstance(api_response, dict): 

753 api_error = api_response.get("error") 

754 error_code = api_error.get("code") 

755 error_message = api_error.get("message") 

756 data = { 

757 'success': False, 

758 'error': "Microsoft Translation API error: Error code {}, {}".format( 

759 error_code, error_message 

760 ), 

761 } 

762 else: 

763 # response body will be of the form: 

764 # [ 

765 # { 

766 # "translations":[ 

767 # {"text": "some chinese text that gave a build error on travis ci", "to": "zh-Hans"}

770 # ] 

771 # for more information, please visit

774 translations = api_response[0].get("translations") 

775 translated_text = translations[0].get("text") 

776 data = {'success': True, 'translation': translated_text} 

777 # catch general connection exception in the requests framework 

778 except requests.exceptions.RequestException as err: 

779 data = { 

780 'success': False, 

781 'error': "Error connecting to Microsoft Translation Service: {0}".format( 

782 err 

783 ), 

784 }

786 return JsonResponse(data)
