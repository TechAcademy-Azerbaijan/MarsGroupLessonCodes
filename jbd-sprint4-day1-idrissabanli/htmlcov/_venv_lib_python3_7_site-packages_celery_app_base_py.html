97class PendingConfiguration(UserDict, AttributeDictMixin):

102 # accessing any key will finalize the configuration,

376 def task(self, *args, **opts): 

377 """Decorator to create a task class out of any callable. 

378 

379 See :ref:`Task options<task-options>` for a list of the 

380 arguments that can be passed to this decorator. 

381 

382 Examples: 

383 .. code-block:: python 

384 

385 @app.task 

386 def refresh_feed(url): 

387 store_feed(feedparser.parse(url)) 

388 

389 with setting extra options: 

390 

391 .. code-block:: python 

392 

393 @app.task(exchange='feeds') 

394 def refresh_feed(url): 

395 return store_feed(feedparser.parse(url)) 

396 

397 Note: 

398 App Binding: For custom apps the task decorator will return 

399 a proxy object, so that the act of creating the task is not 

400 performed until the task is used or the task registry is accessed. 

401 

402 If you're depending on binding to be deferred, then you must 

403 not access any attributes on the returned object until the 

404 application is fully set up (finalized). 

405 """ 

406 if USING_EXECV and opts.get('lazy', True): 

407 # When using execv the task in the original module will point to a 

408 # different app, so doing things like 'add.request' will point to 

409 # a different task instance. This makes sure it will always use 

410 # the task instance from the current app. 

411 # Really need a better solution for this :( 

412 from . import shared_task 

413 return shared_task(*args, lazy=False, **opts)

415 def inner_create_task_cls(shared=True, filter=None, lazy=True, **opts): 

416 _filt = filter 

417 

418 def _create_task_cls(fun): 

419 if shared: 

420 def cons(app): 

421 return app._task_from_fun(fun, **opts) 

422 cons.__name__ = fun.__name__ 

423 connect_on_app_finalize(cons) 

424 if not lazy or self.finalized: 

425 ret = self._task_from_fun(fun, **opts) 

426 else: 

427 # return a proxy object that evaluates on first use 

428 ret = PromiseProxy(self._task_from_fun, (fun,), opts, 

429 __doc__=fun.__doc__) 

430 self._pending.append(ret) 

431 if _filt: 

432 return _filt(ret) 

433 return ret 

434 

435 return _create_task_cls 

436 

437 if len(args) == 1: 

438 if callable(args[0]): 

439 return inner_create_task_cls(**opts)(*args) 

440 raise TypeError('argument 1 to @task() must be a callable') 

441 if args:

444 sum([len(args), len(opts)])))

447 def _task_from_fun(self, fun, name=None, base=None, bind=False, **options): 

448 if not self.finalized and not self.autofinalize: 

449 raise RuntimeError('Contract breach: app not finalized') 

450 name = name or self.gen_task_name(fun.__name__, fun.__module__) 

451 base = base or self.Task 

452 

453 if name not in self._tasks: 

454 run = fun if bind else staticmethod(fun) 

455 task = type(fun.__name__, (base,), dict({ 

456 'app': self, 

457 'name': name, 

458 'run': run, 

459 '_decorated': True, 

460 '__doc__': fun.__doc__, 

461 '__module__': fun.__module__, 

462 '__annotations__': fun.__annotations__, 

463 '__header__': staticmethod(head_from_fun(fun, bound=bind)), 

464 '__wrapped__': run}, **options))() 

465 # for some reason __qualname__ cannot be set in type() 

466 # so we have to set it here. 

467 try: 

468 task.__qualname__ = fun.__qualname__ 

469 except AttributeError: 

470 pass 

471 self._tasks[task.name] = task 

472 task.bind(self) # connects task to this app 

473 add_autoretry_behaviour(task, **options) 

474 else: 

475 task = self._tasks[name] 

476 return task

501 """Finalize the app. 

502 

503 This loads built-in tasks, evaluates pending task decorators, 

504 reads configuration, etc. 

505 """

522 def add_defaults(self, fun): 

523 """Add default configuration from dict ``d``. 

524 

525 If the argument is a callable function then it will be regarded

623 def autodiscover_tasks(self, packages=None, 

624 related_name='tasks', force=False): 

625 """Auto-discover task modules. 

626 

627 Searches a list of packages for a "tasks.py" module (or use 

628 related_name argument). 

629 

630 If the name is empty, this will be delegated to fix-ups (e.g., Django). 

631 

632 For example if you have a directory layout like this: 

633 

634 .. code-block:: text 

635 

636 foo/__init__.py 

637 tasks.py 

638 models.py 

639 

640 bar/__init__.py 

641 tasks.py 

642 models.py 

643 

644 baz/__init__.py 

645 models.py 

646 

647 Then calling ``app.autodiscover_tasks(['foo', 'bar', 'baz'])`` will 

648 result in the modules ``foo.tasks`` and ``bar.tasks`` being imported. 

649 

650 Arguments: 

651 packages (List[str]): List of packages to search. 

652 This argument may also be a callable, in which case the 

653 value returned is used (for lazy evaluation). 

654 related_name (Optional[str]): The name of the module to find. Defaults 

655 to "tasks": meaning "look for 'module.tasks' for every 

656 module in ``packages``.". If ``None`` will only try to import 

657 the package, i.e. "look for 'module'". 

658 force (bool): By default this call is lazy so that the actual 

659 auto-discovery won't happen until an application imports 

660 the default modules. Forcing will cause the auto-discovery 

661 to happen immediately. 

662 """

687 def send_task(self, name, args=None, kwargs=None, countdown=None, 

688 eta=None, task_id=None, producer=None, connection=None, 

689 router=None, result_cls=None, expires=None, 

690 publisher=None, link=None, link_error=None, 

691 add_to_parent=True, group_id=None, group_index=None, 

692 retries=0, chord=None, 

693 reply_to=None, time_limit=None, soft_time_limit=None, 

694 root_id=None, parent_id=None, route_name=None, 

695 shadow=None, chain=None, task_type=None, **options): 

696 """Send task by name. 

697 

698 Supports the same arguments as :meth:`@-Task.apply_async`. 

699 

700 Arguments: 

701 name (str): Name of task to call (e.g., `"tasks.add"`). 

702 result_cls (AsyncResult): Specify custom result class. 

703 """ 

704 parent = have_parent = None 

705 amqp = self.amqp 

706 task_id = task_id or uuid() 

707 producer = producer or publisher # XXX compat 

708 router = router or amqp.router 

709 conf = self.conf 

710 if conf.task_always_eager: # pragma: no cover 

711 warnings.warn(AlwaysEagerIgnored( 

712 'task_always_eager has no effect on send_task', 

713 ), stacklevel=2) 

714 

715 ignored_result = options.pop('ignore_result', False) 

716 options = router.route( 

717 options, route_name or name, args, kwargs, task_type) 

718 

719 if not root_id or not parent_id: 

720 parent = self.current_worker_task 

721 if parent: 

722 if not root_id: 

723 root_id = parent.request.root_id or parent.request.id 

724 if not parent_id: 

725 parent_id = parent.request.id 

726 

727 if conf.task_inherit_parent_priority: 

728 options.setdefault('priority', 

729 parent.request.delivery_info.get('priority')) 

730 

731 message = amqp.create_task_message( 

732 task_id, name, args, kwargs, countdown, eta, group_id, group_index, 

733 expires, retries, chord, 

734 maybe_list(link), maybe_list(link_error), 

735 reply_to or self.thread_oid, time_limit, soft_time_limit, 

736 self.conf.task_send_sent_event, 

737 root_id, parent_id, shadow, chain, 

738 argsrepr=options.get('argsrepr'), 

739 kwargsrepr=options.get('kwargsrepr'), 

740 ) 

741 

742 if connection: 

743 producer = amqp.Producer(connection, auto_declare=False) 

744 

745 with self.producer_or_acquire(producer) as P: 

746 with P.connection._reraise_as_library_errors(): 

747 if not ignored_result: 

748 self.backend.on_task_call(P, task_id) 

749 amqp.send_task_message(P, name, message, **options) 

750 result = (result_cls or self.AsyncResult)(task_id) 

751 # We avoid using the constructor since a custom result class 

752 # can be used, in which case the constructor may still use 

753 # the old signature. 

754 result.ignored = ignored_result 

755 

756 if add_to_parent: 

757 if not have_parent: 

758 parent, have_parent = self.current_worker_task, True 

759 if parent: 

760 parent.add_trail(result) 

761 return result

931 def _load_config(self): 

932 if isinstance(self.on_configure, Signal): 

933 self.on_configure.send(sender=self) 

934 else: 

935 # used to be a method pre 4.0 

936 self.on_configure() 

937 if self._config_source: 

938 self.loader.config_from_object(self._config_source) 

939 self.configured = True 

940 settings = detect_settings( 

941 self.prepare_config(self.loader.conf), self._preconf, 

942 ignore_keys=self._preconf_set_by_auto, prefix=self.namespace, 

943 ) 

944 if self._conf is not None: 

945 # replace in place, as someone may have referenced app.conf, 

946 # done some changes, accessed a key, and then try to make more 

947 # changes to the reference and not the finalized value. 

948 self._conf.swap_with(settings) 

949 else: 

950 self._conf = settings 

951 

952 # load lazy config dict initializers. 

953 pending_def = self._pending_defaults 

954 while pending_def: 

955 self._conf.add_defaults(maybe_evaluate(pending_def.popleft()())) 

956 

957 # load lazy periodic tasks 

958 pending_beat = self._pending_periodic_tasks 

959 while pending_beat: 

960 self._add_periodic_task(*pending_beat.popleft()) 

961 

962 self.on_after_configure.send(sender=self, source=self._conf) 

963 return self._conf

1012 def subclass_with_self(self, Class, name=None, attribute='app', 

1013 reverse=None, keep_reduce=False, **kw): 

1014 """Subclass an app-compatible class. 

1015 

1016 App-compatible means that the class has a class attribute that 

1017 provides the default app it should use, for example:

1019 

1020 Arguments: 

1021 Class (type): The app-compatible class to subclass. 

1022 name (str): Custom name for the target class. 

1023 attribute (str): Name of the attribute holding the app, 

1024 Default is 'app'. 

1025 reverse (str): Reverse path to this object used for pickling 

1026 purposes. For example, to get ``app.AsyncResult``, 

1027 use ``"AsyncResult"``. 

1028 keep_reduce (bool): If enabled a custom ``__reduce__`` 

1029 implementation won't be provided.

1040 __doc__=Class.__doc__, 

1041 **kw) 

1042 if not keep_reduce: 

1043 attrs['__reduce__'] = __reduce__ 

1044 

1045 return type(name or Class.__name__, (Class,), attrs)
