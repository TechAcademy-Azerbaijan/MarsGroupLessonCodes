66 def expiration_timedelta(self): 

67 """Return provider session live seconds. Returns a timedelta ready to 

68 use with session.set_expiry(). 

69 

70 If provider returns a timestamp instead of session seconds to live, the 

71 timedelta is inferred from current time (using UTC timezone). None is 

72 returned if there's no value stored or it's invalid. 

73 """ 

74 if self.extra_data and 'expires' in self.extra_data: 

75 try: 

76 expires = int(self.extra_data.get('expires')) 

77 except (ValueError, TypeError): 

78 return None 

79 

80 now = datetime.utcnow() 

81 

82 # Detect if expires is a timestamp 

83 if expires > time.mktime(now.timetuple()): 

84 # expires is a datetime, return the remaining difference 

85 return datetime.utcfromtimestamp(expires) - now 

86 else: 

87 # expires is the time to live seconds since creation, 

88 # check against auth_time if present, otherwise return 

89 # the value 

90 auth_time = self.extra_data.get('auth_time') 

91 if auth_time: 

92 reference = datetime.utcfromtimestamp(auth_time) 

93 return (reference + timedelta(seconds=expires)) - now 

94 else: 

95 return timedelta(seconds=expires)
