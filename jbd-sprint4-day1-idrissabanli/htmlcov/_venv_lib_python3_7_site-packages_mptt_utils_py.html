16def previous_current_next(items): 

17 """ 

18 From http://www.wordaligned.org/articles/zippy-triples-served-with-python 

19 

20 Creates an iterator which returns (previous, current, next) triples, 

21 with ``None`` filling in when there is no previous or next 

22 available. 

23 """ 

24 extend = itertools.chain([None], items, [None]) 

25 prev, cur, nex = itertools.tee(extend, 3) 

26 # Advancing an iterator twice when we know there are two items (the 

27 # two Nones at the start and at the end) will never fail except if 

28 # `items` is some funny StopIteration-raising generator. There's no point 

29 # in swallowing this exception. 

30 next(cur) 

31 next(nex) 

32 next(nex) 

33 return zip(prev, cur, nex)

36def tree_item_iterator(items, ancestors=False, callback=str): 

37 """ 

38 Given a list of tree items, iterates over the list, generating 

39 two-tuples of the current tree item and a ``dict`` containing 

40 information about the tree structure around the item, with the 

41 following keys: 

42 

43 ``'new_level'`` 

44 ``True`` if the current item is the start of a new level in 

45 the tree, ``False`` otherwise. 

46 

47 ``'closed_levels'`` 

48 A list of levels which end after the current item. This will 

49 be an empty list if the next item is at the same level as the 

50 current item. 

51 

52 If ``ancestors`` is ``True``, the following key will also be 

53 available: 

54 

55 ``'ancestors'`` 

56 A list of representations of the ancestors of the current 

57 node, in descending order (root node first, immediate parent 

58 last). 

59 

60 For example: given the sample tree below, the contents of the 

61 list which would be available under the ``'ancestors'`` key 

62 are given on the right:: 

63 

64 Books -> [] 

65 Sci-fi -> [u'Books'] 

66 Dystopian Futures -> [u'Books', u'Sci-fi'] 

67 

68 You can overload the default representation by providing an 

69 optional ``callback`` function which takes a single argument 

70 and performs coersion as required. 

71 

72 """ 

73 structure = {} 

74 opts = None 

75 first_item_level = 0 

76 for previous, current, next_ in previous_current_next(items): 

77 if opts is None: 

78 opts = current._mptt_meta 

79 

80 current_level = getattr(current, opts.level_attr) 

81 if previous: 

82 structure['new_level'] = (getattr(previous, 

83 opts.level_attr) < current_level) 

84 if ancestors: 

85 # If the previous node was the end of any number of 

86 # levels, remove the appropriate number of ancestors 

87 # from the list. 

88 if structure['closed_levels']: 

89 structure['ancestors'] = \ 

90 structure['ancestors'][:-len(structure['closed_levels'])] 

91 # If the current node is the start of a new level, add its 

92 # parent to the ancestors list. 

93 if structure['new_level']: 

94 structure['ancestors'].append(callback(previous)) 

95 else: 

96 structure['new_level'] = True 

97 if ancestors: 

98 # Set up the ancestors list on the first item 

99 structure['ancestors'] = [] 

100 

101 first_item_level = current_level 

102 if next_: 

103 structure['closed_levels'] = list(range( 

104 current_level, 

105 getattr(next_, opts.level_attr), 

106 -1)) 

107 else: 

108 # All remaining levels need to be closed 

109 structure['closed_levels'] = list(range( 

110 current_level, 

111 first_item_level - 1, 

112 -1)) 

113 

114 # Return a deep copy of the structure dict so this function can 

115 # be used in situations where the iterator is consumed 

116 # immediately. 

117 yield current, copy.deepcopy(structure)

120def drilldown_tree_for_node(node, rel_cls=None, rel_field=None, count_attr=None, 

121 cumulative=False, all_descendants=False): 

122 """ 

123 Creates a drilldown tree for the given node. A drilldown tree 

124 consists of a node's ancestors, itself and its immediate children 

125 or all descendants, all in tree order. 

126 

127 Optional arguments may be given to specify a ``Model`` class which 

128 is related to the node's class, for the purpose of adding related 

129 item counts to the node's children: 

130 

131 ``rel_cls`` 

132 A ``Model`` class which has a relation to the node's class. 

133 

134 ``rel_field`` 

135 The name of the field in ``rel_cls`` which holds the relation 

136 to the node's class. 

137 

138 ``count_attr`` 

139 The name of an attribute which should be added to each child in 

140 the drilldown tree, containing a count of how many instances 

141 of ``rel_cls`` are related through ``rel_field``. 

142 

143 ``cumulative`` 

144 If ``True``, the count will be for each child and all of its 

145 descendants, otherwise it will be for each child itself. 

146 

147 ``all_descendants`` 

148 If ``True``, return all descendants, not just immediate children. 

149 """ 

150 if all_descendants: 

151 children = node.get_descendants() 

152 else: 

153 children = node.get_children() 

154 if rel_cls and rel_field and count_attr: 

155 children = node._tree_manager.add_related_count( 

156 children, rel_cls, rel_field, count_attr, cumulative) 

157 return itertools.chain(node.get_ancestors(), [node], children)

160def print_debug_info(qs, file=None): 

161 """ 

162 Given an mptt queryset, prints some debug information to stdout. 

163 Use this when things go wrong. 

164 Please include the output from this method when filing bug issues. 

165 """ 

166 opts = qs.model._mptt_meta 

167 writer = csv.writer(sys.stdout if file is None else file) 

168 header = ( 

169 'pk', 

170 opts.level_attr, 

171 '%s_id' % opts.parent_attr, 

172 opts.tree_id_attr, 

173 opts.left_attr, 

174 opts.right_attr, 

175 'pretty', 

176 ) 

177 writer.writerow(header) 

178 for n in qs.order_by('tree_id', 'lft'): 

179 level = getattr(n, opts.level_attr) 

180 row = [] 

181 for field in header[:-1]: 

182 row.append(getattr(n, field)) 

183 

184 row_text = '%s%s' % ('- ' * level, str(n)) 

185 row.append(row_text) 

186 writer.writerow(row)

189def _get_tree_model(model_class): 

190 # Find the model that contains the tree fields. 

191 # This is a weird way of going about it, but Django doesn't let us access 

192 # the fields list to detect where the tree fields actually are, 

193 # because the app cache hasn't been loaded yet. 

194 # So, it *should* be the *last* concrete MPTTModel subclass in the mro(). 

195 bases = list(model_class.mro()) 

196 while bases: 

197 b = bases.pop() 

198 # NOTE can't use `issubclass(b, MPTTModel)` here because we can't import MPTTModel yet! 

199 # So hasattr(b, '_mptt_meta') will have to do. 

200 if hasattr(b, '_mptt_meta') and not (b._meta.abstract or b._meta.proxy): 

201 return b 

202 return None

205def get_cached_trees(queryset): 

206 """ 

207 Takes a list/queryset of model objects in MPTT left (depth-first) order and 

208 caches the children and parent on each node. This allows up and down 

209 traversal through the tree without the need for further queries. Use cases 

210 include using a recursively included template or arbitrarily traversing 

211 trees. 

212 

213 NOTE: nodes _must_ be passed in the correct (depth-first) order. If they aren't, 

214 a ValueError will be raised. 

215 

216 Returns a list of top-level nodes. If a single tree was provided in its 

217 entirety, the list will of course consist of just the tree's root node. 

218 

219 For filtered querysets, if no ancestors for a node are included in the 

220 queryset, it will appear in the returned list as a top-level node. 

221 

222 Aliases to this function are also available: 

223 

224 ``mptt.templatetags.mptt_tag.cache_tree_children`` 

225 Use for recursive rendering in templates. 

226 

227 ``mptt.querysets.TreeQuerySet.get_cached_trees`` 

228 Useful for chaining with queries; e.g., 

229 `Node.objects.filter(**kwargs).get_cached_trees()` 

230 """ 

231 

232 current_path = [] 

233 top_nodes = [] 

234 

235 if queryset: 

236 # Get the model's parent-attribute name 

237 parent_attr = queryset[0]._mptt_meta.parent_attr 

238 root_level = None 

239 is_filtered = (hasattr(queryset, "query") and queryset.query.has_filters()) 

240 for obj in queryset: 

241 # Get the current mptt node level 

242 node_level = obj.get_level() 

243 

244 if root_level is None or (is_filtered and node_level < root_level): 

245 # First iteration, so set the root level to the top node level 

246 root_level = node_level 

247 

248 elif node_level < root_level: 

249 # ``queryset`` was a list or other iterable (unable to order), 

250 # and was provided in an order other than depth-first 

251 raise ValueError( 

252 _('Node %s not in depth-first order') % (type(queryset),) 

253 ) 

254 

255 # Set up the attribute on the node that will store cached children, 

256 # which is used by ``MPTTModel.get_children`` 

257 obj._cached_children = [] 

258 

259 # Remove nodes not in the current branch 

260 while len(current_path) > node_level - root_level: 

261 current_path.pop(-1) 

262 

263 if node_level == root_level: 

264 # Add the root to the list of top nodes, which will be returned 

265 top_nodes.append(obj) 

266 else: 

267 # Cache the parent on the current node, and attach the current 

268 # node to the parent's list of children 

269 _parent = current_path[-1] 

270 setattr(obj, parent_attr, _parent) 

271 _parent._cached_children.append(obj) 

272 

273 if root_level == 0: 

274 # get_ancestors() can use .parent.parent.parent... 

275 setattr(obj, '_mptt_use_cached_ancestors', True) 

276 

277 # Add the current node to end of the current path - the last node 

278 # in the current path is the parent for the next iteration, unless 

279 # the next iteration is higher up the tree (a new branch), in which 

280 # case the paths below it (e.g., this one) will be removed from the 

281 # current path during the next iteration 

282 current_path.append(obj) 

283 

284 return top_nodes
