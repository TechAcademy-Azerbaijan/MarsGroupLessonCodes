51class LocalTimezone(tzinfo): 

52 """Local time implementation. 

53 

54 Note: 

55 Used only when the :setting:`enable_utc` setting is disabled. 

56 """ 

57 

58 _offset_cache = {} 

59 

60 def __init__(self): 

61 # This code is moved in __init__ to execute it as late as possible 

62 # See get_default_timezone(). 

63 self.STDOFFSET = timedelta(seconds=-_time.timezone) 

64 if _time.daylight: 

65 self.DSTOFFSET = timedelta(seconds=-_time.altzone) 

66 else: 

67 self.DSTOFFSET = self.STDOFFSET 

68 self.DSTDIFF = self.DSTOFFSET - self.STDOFFSET 

69 tzinfo.__init__(self) 

70 

71 def __repr__(self): 

72 return f'<LocalTimezone: UTC{int(self.DSTOFFSET.total_seconds() / 3600):+03d}>' 

73 

74 def utcoffset(self, dt): 

75 return self.DSTOFFSET if self._isdst(dt) else self.STDOFFSET 

76 

77 def dst(self, dt): 

78 return self.DSTDIFF if self._isdst(dt) else ZERO 

79 

80 def tzname(self, dt): 

81 return _time.tzname[self._isdst(dt)]

83 def fromutc(self, dt): 

84 # The base tzinfo class no longer implements a DST 

85 # offset aware .fromutc() in Python 3 (Issue #2306). 

86 

87 # I'd rather rely on pytz to do this, than port 

88 # the C code from cpython's fromutc [asksol] 

89 offset = int(self.utcoffset(dt).seconds / 60.0) 

90 try: 

91 tz = self._offset_cache[offset] 

92 except KeyError: 

93 tz = self._offset_cache[offset] = FixedOffset(offset) 

94 return tz.fromutc(dt.replace(tzinfo=tz))

96 def _isdst(self, dt): 

97 tt = (dt.year, dt.month, dt.day, 

98 dt.hour, dt.minute, dt.second, 

99 dt.weekday(), 0, 0) 

100 stamp = _time.mktime(tt) 

101 tt = _time.localtime(stamp) 

102 return tt.tm_isdst > 0

152def delta_resolution(dt, delta): 

153 """Round a :class:`~datetime.datetime` to the resolution of timedelta. 

154 

155 If the :class:`~datetime.timedelta` is in days, the 

156 :class:`~datetime.datetime` will be rounded to the nearest days, 

157 if the :class:`~datetime.timedelta` is in hours the 

158 :class:`~datetime.datetime` will be rounded to the nearest hour, 

159 and so on until seconds, which will just return the original 

160 :class:`~datetime.datetime`. 

161 """ 

162 delta = max(delta.total_seconds(), 0) 

163 

164 resolutions = ((3, lambda x: x / 86400), 

165 (4, lambda x: x / 3600), 

166 (5, lambda x: x / 60)) 

167 

168 args = dt.year, dt.month, dt.day, dt.hour, dt.minute, dt.second 

169 for res, predicate in resolutions: 

170 if predicate(delta) >= 1.0: 

171 return datetime(*args[:res], tzinfo=dt.tzinfo) 

172 return dt

175def remaining(start, ends_in, now=None, relative=False): 

176 """Calculate the remaining time for a start date and a timedelta. 

177 

178 For example, "how many seconds left for 30 seconds after start?" 

179 

180 Arguments: 

181 start (~datetime.datetime): Starting date. 

182 ends_in (~datetime.timedelta): The end delta. 

183 relative (bool): If enabled the end time will be calculated 

184 using :func:`delta_resolution` (i.e., rounded to the 

185 resolution of `ends_in`). 

186 now (Callable): Function returning the current time and date. 

187 Defaults to :func:`datetime.utcnow`. 

188 

189 Returns: 

190 ~datetime.timedelta: Remaining time. 

191 """ 

192 now = now or datetime.utcnow() 

193 if str(start.tzinfo) == str(now.tzinfo) and now.utcoffset() != start.utcoffset(): 

194 # DST started/ended 

195 start = start.replace(tzinfo=now.tzinfo) 

196 end_date = start + ends_in 

197 if relative: 

198 end_date = delta_resolution(end_date, ends_in).replace(microsecond=0) 

199 ret = end_date - now 

200 if C_REMDEBUG: # pragma: no cover 

201 print('rem: NOW:{!r} START:{!r} ENDS_IN:{!r} END_DATE:{} REM:{}'.format( 

202 now, start, ends_in, end_date, ret)) 

203 return ret

231def humanize_seconds(secs, prefix='', sep='', now='now', microseconds=False): 

232 """Show seconds in human form. 

233 

234 For example, 60 becomes "1 minute", and 7200 becomes "2 hours". 

235 

236 Arguments: 

237 prefix (str): can be used to add a preposition to the output 

238 (e.g., 'in' will give 'in 1 second', but add nothing to 'now'). 

239 now (str): Literal 'now'. 

240 microseconds (bool): Include microseconds. 

241 """ 

242 secs = float(format(float(secs), '.2f')) 

243 for unit, divider, formatter in TIME_UNITS: 

244 if secs >= divider: 

245 w = secs / float(divider) 

246 return '{}{}{} {}'.format(prefix, sep, formatter(w), 

247 pluralize(w, unit)) 

248 if microseconds and secs > 0.0: 

249 return '{prefix}{sep}{0:.2f} seconds'.format( 

250 secs, sep=sep, prefix=prefix) 

251 return now

268def make_aware(dt, tz): 

269 """Set timezone for a :class:`~datetime.datetime` object.""" 

270 try: 

271 _localize = tz.localize 

272 except AttributeError:

283def localize(dt, tz): 

284 """Convert aware :class:`~datetime.datetime` to another timezone.""" 

285 if is_naive(dt): # Ensure timezone aware datetime 

286 dt = make_aware(dt, tz) 

287 if dt.tzinfo == utc: 

288 dt = dt.astimezone(tz) # Always safe to call astimezone on utc zones 

289 try: 

290 _normalize = tz.normalize 

291 except AttributeError: # non-pytz tz 

292 return dt 

293 else: 

294 try: 

295 return _normalize(dt, is_dst=None) 

296 except TypeError: 

297 return _normalize(dt) 

298 except AmbiguousTimeError: 

299 return min(_normalize(dt, is_dst=True), 

300 _normalize(dt, is_dst=False))

318class ffwd: 

319 """Version of ``dateutil.relativedelta`` that only supports addition.""" 

320 

321 def __init__(self, year=None, month=None, weeks=0, weekday=None, day=None, 

322 hour=None, minute=None, second=None, microsecond=None, 

323 **kwargs): 

324 # pylint: disable=redefined-outer-name 

325 # weekday is also a function in outer scope. 

326 self.year = year 

327 self.month = month 

328 self.weeks = weeks 

329 self.weekday = weekday 

330 self.day = day 

331 self.hour = hour 

332 self.minute = minute 

333 self.second = second 

334 self.microsecond = microsecond 

335 self.days = weeks * 7 

336 self._has_time = self.hour is not None or self.minute is not None 

337 

338 def __repr__(self): 

339 return reprcall('ffwd', (), self._fields(weeks=self.weeks, 

340 weekday=self.weekday)) 

341 

342 def __radd__(self, other): 

343 if not isinstance(other, date): 

344 return NotImplemented 

345 year = self.year or other.year 

346 month = self.month or other.month 

347 day = min(monthrange(year, month)[1], self.day or other.day) 

348 ret = other.replace(**dict(dictfilter(self._fields()), 

349 year=year, month=month, day=day)) 

350 if self.weekday is not None: 

351 ret += timedelta(days=(7 - ret.weekday() + self.weekday) % 7) 

352 return ret + timedelta(days=self.days)

355 return dictfilter({

357 'hour': self.hour, 'minute': self.minute, 

358 'second': self.second, 'microsecond': self.microsecond,

374def get_exponential_backoff_interval( 

375 factor, 

376 retries, 

377 maximum, 

378 full_jitter=False 

379): 

380 """Calculate the exponential backoff wait time.""" 

381 # Will be zero if factor equals 0 

382 countdown = min(maximum, factor * (2 ** retries)) 

383 # Full jitter according to 

384 # https://www.awsarchitectureblog.com/2015/03/backoff.html 

385 if full_jitter: 

386 countdown = random.randrange(countdown + 1) 

387 # Adjust according to maximum wait time and account for negative values. 

388 return max(0, countdown)
