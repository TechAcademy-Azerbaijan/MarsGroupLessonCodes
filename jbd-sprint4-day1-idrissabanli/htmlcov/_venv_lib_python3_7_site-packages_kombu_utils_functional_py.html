44class LRUCache(UserDict): 

45 """LRU Cache implementation using a doubly linked list to track access. 

46 

47 Arguments: 

48 limit (int): The maximum number of keys to keep in the cache. 

49 When a new key is inserted and the limit has been exceeded, 

50 the *Least Recently Used* key will be discarded from the 

51 cache. 

52 """ 

53 

54 def __init__(self, limit=None): 

55 self.limit = limit 

56 self.mutex = threading.RLock() 

57 self.data = OrderedDict() 

58 

59 def __getitem__(self, key): 

60 with self.mutex: 

61 value = self[key] = self.data.pop(key) 

62 return value 

63 

64 def update(self, *args, **kwargs): 

65 with self.mutex: 

66 data, limit = self.data, self.limit 

67 data.update(*args, **kwargs) 

68 if limit and len(data) > limit: 

69 # pop additional items in case limit exceeded 

70 for _ in range(len(data) - limit): 

71 data.popitem(last=False) 

72 

73 def popitem(self, last=True): 

74 with self.mutex: 

75 return self.data.popitem(last) 

76 

77 def __setitem__(self, key, value): 

78 # remove least recently used key. 

79 with self.mutex: 

80 if self.limit and len(self.data) >= self.limit: 

81 self.data.pop(next(iter(self.data))) 

82 self.data[key] = value 

83 

84 def __iter__(self): 

85 return iter(self.data) 

86 

87 def _iterate_items(self): 

88 with self.mutex: 

89 for k in self: 

90 try: 

91 yield (k, self.data[k]) 

92 except KeyError: # pragma: no cover 

93 pass 

94 iteritems = _iterate_items 

95 

96 def _iterate_values(self): 

97 with self.mutex: 

98 for k in self: 

99 try: 

100 yield self.data[k] 

101 except KeyError: # pragma: no cover 

102 pass 

103 

104 itervalues = _iterate_values 

105 

106 def _iterate_keys(self): 

107 # userdict.keys in py3k calls __getitem__ 

108 with self.mutex: 

109 return self.data.keys() 

110 iterkeys = _iterate_keys 

111 

112 def incr(self, key, delta=1): 

113 with self.mutex: 

114 # this acts as memcached does- store as a string, but return a 

115 # integer as long as it exists and we can cast it 

116 newval = int(self.data.pop(key)) + delta 

117 self[key] = str(newval) 

118 return newval

120 def __getstate__(self): 

121 d = dict(vars(self)) 

122 d.pop('mutex') 

123 return d

126 self.__dict__ = state 

127 self.mutex = threading.RLock()

129 if sys.version_info[0] == 3: # pragma: no cover

145def memoize(maxsize=None, keyfun=None, Cache=LRUCache): 

146 """Decorator to cache function return value.""" 

147 def _memoize(fun): 

148 mutex = threading.Lock() 

149 cache = Cache(limit=maxsize) 

150 

151 @wraps(fun) 

152 def _M(*args, **kwargs): 

153 if keyfun: 

154 key = keyfun(args, kwargs) 

155 else: 

156 key = args + (KEYWORD_MARK,) + tuple(sorted(kwargs.items())) 

157 try: 

158 with mutex: 

159 value = cache[key] 

160 except KeyError: 

161 value = fun(*args, **kwargs) 

162 _M.misses += 1 

163 with mutex: 

164 cache[key] = value 

165 else: 

166 _M.hits += 1 

167 return value 

168 

169 def clear(): 

170 """Clear the cache and reset cache statistics.""" 

171 cache.clear() 

172 _M.hits = _M.misses = 0 

173 

174 _M.hits = _M.misses = 0 

175 _M.clear = clear 

176 _M.original_func = fun 

177 return _M 

178 

179 return _memoize

259def fxrange(start=1.0, stop=None, step=1.0, repeatlast=False): 

260 cur = start * 1.0 

261 while 1: 

262 if not stop or cur <= stop: 

263 yield cur 

264 cur += step 

265 else: 

266 if not repeatlast: 

267 break

284def retry_over_time(fun, catch, args=None, kwargs=None, errback=None, 

285 max_retries=None, interval_start=2, interval_step=2, 

286 interval_max=30, callback=None, timeout=None): 

287 """Retry the function over and over until max retries is exceeded. 

288 

289 For each retry we sleep a for a while before we try again, this interval 

290 is increased for every retry until the max seconds is reached. 

291 

292 Arguments: 

293 fun (Callable): The function to try 

294 catch (Tuple[BaseException]): Exceptions to catch, can be either 

295 tuple or a single exception class. 

296 

297 Keyword Arguments: 

298 args (Tuple): Positional arguments passed on to the function. 

299 kwargs (Dict): Keyword arguments passed on to the function. 

300 errback (Callable): Callback for when an exception in ``catch`` 

301 is raised. The callback must take three arguments: 

302 ``exc``, ``interval_range`` and ``retries``, where ``exc`` 

303 is the exception instance, ``interval_range`` is an iterator 

304 which return the time in seconds to sleep next, and ``retries`` 

305 is the number of previous retries. 

306 max_retries (int): Maximum number of retries before we give up. 

307 If neither of this and timeout is set, we will retry forever. 

308 If one of this and timeout is reached, stop. 

309 interval_start (float): How long (in seconds) we start sleeping 

310 between retries. 

311 interval_step (float): By how much the interval is increased for 

312 each retry. 

313 interval_max (float): Maximum number of seconds to sleep 

314 between retries. 

315 timeout (int): Maximum seconds waiting before we give up. 

316 """ 

317 kwargs = {} if not kwargs else kwargs 

318 args = [] if not args else args 

319 interval_range = fxrange(interval_start, 

320 interval_max + interval_start, 

321 interval_step, repeatlast=True) 

322 end = time() + timeout if timeout else None 

323 for retries in count(): 

324 try: 

325 return fun(*args, **kwargs) 

326 except catch as exc: 

327 if max_retries is not None and retries >= max_retries: 

328 raise 

329 if end and time() > end: 

330 raise 

331 if callback: 

332 callback() 

333 tts = float(errback(exc, interval_range, retries) if errback 

334 else next(interval_range)) 

335 if tts: 

336 for _ in range(int(tts)): 

337 if callback: 

338 callback() 

339 sleep(1.0) 

340 # sleep remainder after int truncation above. 

341 sleep(abs(int(tts) - tts))
