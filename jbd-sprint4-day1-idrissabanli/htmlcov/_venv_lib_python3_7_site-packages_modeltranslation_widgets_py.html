9class ClearableWidgetWrapper(Widget): 

10 """ 

11 Wraps another widget adding a clear checkbox, making it possible to 

12 reset the field to some empty value even if the original input doesn't 

13 have means to. 

14 

15 Useful for ``TextInput`` and ``Textarea`` based widgets used in combination 

16 with nullable text fields. 

17 

18 Use it in ``Field.formfield`` or ``ModelAdmin.formfield_for_dbfield``: 

19 

20 field.widget = ClearableWidgetWrapper(field.widget) 

21 

22 ``None`` is assumed to be a proper choice for the empty value, but you may 

23 pass another one to the constructor. 

24 """ 

25 clear_checkbox_label = _("None") 

26 template = '<span class="clearable-input">{0} <span>{2}</span> {3}</span>'

34 def __init__(self, widget, empty_value=None): 

35 """ 

36 Remebers the widget we are wrapping and precreates a checkbox input. 

37 Allows overriding the empty value. 

38 """ 

39 self.widget = widget 

40 self.checkbox = CheckboxInput(attrs={'tabindex': '-1'}) 

41 self.empty_value = empty_value 

42 

43 def __getattr__(self, name): 

44 """ 

45 If we don't have a property or a method, chances are the wrapped 

46 widget does. 

47 """ 

48 if name != 'widget': 

49 return getattr(self.widget, name) 

50 raise AttributeError 

51 

52 @property 

53 def media(self): 

54 """ 

55 Combines media of both components and adds a small script that unchecks 

56 the clear box, when a value in any wrapped input is modified. 

57 """ 

58 return self.widget.media + self.checkbox.media + Media(self.Media) 

59 

60 def render(self, name, value, attrs=None, renderer=None): 

61 """ 

62 Appends a checkbox for clearing the value (that is, setting the field 

63 with the ``empty_value``). 

64 """ 

65 wrapped = self.widget.render(name, value, attrs, renderer) 

66 checkbox_name = self.clear_checkbox_name(name) 

67 checkbox_id = self.clear_checkbox_id(checkbox_name) 

68 checkbox_label = self.clear_checkbox_label 

69 checkbox = self.checkbox.render( 

70 checkbox_name, 

71 value == self.empty_value, 

72 attrs={'id': checkbox_id}, 

73 renderer=renderer) 

74 return mark_safe(self.template.format( 

75 conditional_escape(wrapped), 

76 conditional_escape(checkbox_id), 

77 conditional_escape(checkbox_label), 

78 conditional_escape(checkbox)))

80 def value_from_datadict(self, data, files, name): 

81 """ 

82 If the clear checkbox is checked returns the configured empty value, 

83 completely ignoring the original input. 

84 """ 

85 clear = self.checkbox.value_from_datadict(data, files, self.clear_checkbox_name(name)) 

86 if clear: 

87 return self.empty_value 

88 return self.widget.value_from_datadict(data, files, name)
