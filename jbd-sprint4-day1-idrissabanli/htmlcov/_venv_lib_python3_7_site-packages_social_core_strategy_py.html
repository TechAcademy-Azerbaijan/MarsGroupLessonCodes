62 def openid_session_dict(self, name):

64 # to JSON to avoid code execution risks. Flask did this from Flask 

65 # 0.10, Django is switching to JSON by default from version 1.6.

110 def random_string(self, length=12, chars=ALLOWED_CHARS): 

111 # Implementation borrowed from django 1.4 

112 try: 

113 random.SystemRandom() 

114 except NotImplementedError: 

115 key = self.setting('SECRET_KEY', '') 

116 seed = '{0}{1}{2}'.format(random.getstate(), time.time(), key) 

117 random.seed(hashlib.sha256(seed.encode()).digest()) 

118 return ''.join([random.choice(chars) for i in range(length)])
