41def by_name(backend=None, loader=None, 

42 extension_namespace='celery.result_backends'): 

43 """Get backend class by name/alias.""" 

44 backend = backend or 'disabled' 

45 loader = loader or current_app.loader 

46 aliases = dict(BACKEND_ALIASES, **loader.override_backends) 

47 aliases.update( 

48 load_extension_class_names(extension_namespace) or {}) 

49 try: 

50 cls = symbol_by_name(backend, aliases) 

51 except ValueError as exc: 

52 reraise(ImproperlyConfigured, ImproperlyConfigured( 

53 UNKNOWN_BACKEND.strip().format(backend, exc)), sys.exc_info()[2]) 

54 if isinstance(cls, types.ModuleType): 

55 raise ImproperlyConfigured(UNKNOWN_BACKEND.strip().format( 

56 backend, 'is a Python module, not a backend class.')) 

57 return cls

60def by_url(backend=None, loader=None): 

61 """Get backend class by URL.""" 

62 url = None 

63 if backend and '://' in backend: 

64 url = backend 

65 scheme, _, _ = url.partition('://') 

66 if '+' in scheme: 

67 backend, url = url.split('+', 1) 

68 else: 

69 backend = scheme 

70 return by_name(backend, loader), url
