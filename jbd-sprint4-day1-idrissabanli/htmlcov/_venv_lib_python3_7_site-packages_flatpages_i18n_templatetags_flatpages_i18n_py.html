69def get_flatpages_i18n(parser, token): 

70 """ 

71 Retrieves all flatpage objects available for the current site and 

72 visible to the specific user (or visible to all users if no user is 

73 specified). Populates the template context with them in a variable 

74 whose name is defined by the ``as`` clause. 

75 

76 An optional ``for`` clause can be used to control the user whose 

77 permissions are to be used in determining which flatpages are visible. 

78 

79 An optional argument, ``starts_with``, can be applied to limit the 

80 returned flatpages to those beginning with a particular base URL. 

81 This argument can be passed as a variable or a string, as it resolves 

82 from the template context. 

83 

84 Syntax:: 

85 

86 {% get_flatpages_i18n ['url_starts_with'] [for user] as context_name %} 

87 

88 Example usage:: 

89 

90 {% get_flatpages_i18n as flatpages %} 

91 {% get_flatpages_i18n for someuser as flatpages %} 

92 {% get_flatpages_i18n '/about/' as about_pages %} 

93 {% get_flatpages_i18n prefix as about_pages %} 

94 {% get_flatpages_i18n '/about/' for someuser as about_pages %} 

95 {% get_flatpages_i18n containing '/en/' as my_pages %} 

96 {% get_flatpages_i18n excluding '/en/' as my_pages %} 

97 """ 

98 bits = token.split_contents() 

99 syntax_message = ("%(tag_name)s expects a syntax of %(tag_name)s " 

100 "['url_starts_with'] [for user] as context_name" % 

101 dict(tag_name=bits[0])) 

102 

103 # Must have at 3-6 bits in the tag 

104 if 3 <= len(bits) <= 8: 

105 containing = None 

106 excluding = None 

107 

108 # If there's an even number of bits, there's no prefix 

109 if len(bits) % 2 == 0: 

110 prefix = bits[1] 

111 else: 

112 prefix = None 

113 if bits[1] == "containing": 

114 containing = bits[2] 

115 else: 

116 if bits[1] == "excluding": 

117 excluding = bits[2] 

118 

119 # The very last bit must be the context name 

120 if bits[-2] != 'as': 

121 raise template.TemplateSyntaxError(syntax_message) 

122 context_name = bits[-1] 

123 

124 # If there are 5 or 6 bits, there is a user defined 

125 if len(bits) >= 5: 

126 if bits[-4] != 'for': 

127 raise template.TemplateSyntaxError(syntax_message) 

128 user = bits[-3] 

129 else: 

130 user = None 

131 

132 return FlatpageNode( 

133 context_name, starts_with=prefix, contains=containing, 

134 excludes=excluding, user=user) 

135 else: 

136 raise template.TemplateSyntaxError(syntax_message)
