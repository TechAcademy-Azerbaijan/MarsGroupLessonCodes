13def get_language(): 

14 """ 

15 Return an active language code that is guaranteed to be in 

16 settings.LANGUAGES (Django does not seem to guarantee this for us). 

17 """ 

18 lang = _get_language() 

19 if lang is None: # Django >= 1.8 

20 return settings.DEFAULT_LANGUAGE 

21 if lang not in settings.AVAILABLE_LANGUAGES and '-' in lang: 

22 lang = lang.split('-')[0] 

23 if lang in settings.AVAILABLE_LANGUAGES: 

24 return lang 

25 return settings.DEFAULT_LANGUAGE

60def _join_css_class(bits, offset): 

61 if '-'.join(bits[-offset:]) in settings.AVAILABLE_LANGUAGES + ['en-us']: 

62 return '%s-%s' % ('_'.join(bits[:len(bits) - offset]), '_'.join(bits[-offset:]))

66def build_css_class(localized_fieldname, prefix=''): 

67 """ 

68 Returns a css class based on ``localized_fieldname`` which is easily 

69 splittable and capable of regionalized language codes. 

70 

71 Takes an optional ``prefix`` which is prepended to the returned string. 

72 """ 

73 bits = localized_fieldname.split('_') 

74 css_class = '' 

75 if len(bits) == 1: 

76 css_class = str(localized_fieldname) 

77 elif len(bits) == 2: 

78 # Fieldname without underscore and short language code 

79 # Examples: 

80 # 'foo_de' --> 'foo-de', 

81 # 'bar_en' --> 'bar-en' 

82 css_class = '-'.join(bits) 

83 elif len(bits) > 2: 

84 # Try regionalized language code 

85 # Examples: 

86 # 'foo_es_ar' --> 'foo-es_ar', 

87 # 'foo_bar_zh_tw' --> 'foo_bar-zh_tw' 

88 css_class = _join_css_class(bits, 2) 

89 if not css_class: 

90 # Try short language code 

91 # Examples: 

92 # 'foo_bar_de' --> 'foo_bar-de', 

93 # 'foo_bar_baz_de' --> 'foo_bar_baz-de' 

94 css_class = _join_css_class(bits, 1) 

95 return '%s-%s' % (prefix, css_class) if prefix else css_class

98def unique(seq): 

99 """ 

100 Returns a generator yielding unique sequence members in order 

101 

102 A set by itself will return unique values without any regard for order. 

103 

104 >>> list(unique([1, 2, 3, 2, 2, 4, 1])) 

105 [1, 2, 3, 4] 

106 """ 

107 seen = set() 

108 return (x for x in seq if x not in seen and not seen.add(x))

111def resolution_order(lang, override=None): 

112 """ 

113 Return order of languages which should be checked for parameter language. 

114 First is always the parameter language, later are fallback languages. 

115 Override parameter has priority over FALLBACK_LANGUAGES. 

116 """ 

117 if not settings.ENABLE_FALLBACKS: 

118 return (lang,) 

119 if override is None: 

120 override = {} 

121 fallback_for_lang = override.get(lang, settings.FALLBACK_LANGUAGES.get(lang, ())) 

122 fallback_def = override.get('default', settings.FALLBACK_LANGUAGES['default']) 

123 order = (lang,) + fallback_for_lang + fallback_def 

124 return tuple(unique(order))

127@contextmanager 

128def auto_populate(mode='all'): 

129 """ 

130 Overrides translation fields population mode (population mode decides which 

131 unprovided translations will be filled during model construction / loading). 

132 

133 Example: 

134 

135 with auto_populate('all'): 

136 s = Slugged.objects.create(title='foo') 

137 s.title_en == 'foo' // True 

138 s.title_de == 'foo' // True 

139 

140 This method may be used to ensure consistency loading untranslated fixtures, 

141 with non-default language active: 

142 

143 with auto_populate('required'): 

144 call_command('loaddata', 'fixture.json') 

145 """ 

146 current_population_mode = settings.AUTO_POPULATE 

147 settings.AUTO_POPULATE = mode 

148 try: 

149 yield 

150 finally: 

151 settings.AUTO_POPULATE = current_population_mode
