130 def is_due(self, last_run_at): 

131 """Return tuple of ``(is_due, next_time_to_check)``. 

132 

133 Notes: 

134 - next time to check is in seconds. 

135 

136 - ``(True, 20)``, means the task should be run now, and the next 

137 time to check is in 20 seconds. 

138 

139 - ``(False, 12.3)``, means the task is not due, but that the 

140 scheduler should check again in 12.3 seconds. 

141 

142 The next time to check is used to save energy/CPU cycles, 

143 it does not need to be accurate but will influence the precision 

144 of your schedule. You must also keep in mind 

145 the value of :setting:`beat_max_loop_interval`, 

146 that decides the maximum number of seconds the scheduler can 

147 sleep between re-checking the periodic task intervals. So if you 

148 have a task that changes schedule at run-time then your next_run_at 

149 check will decide how long it will take before a change to the 

150 schedule takes effect. The max loop interval takes precedence 

151 over the next check at value returned. 

152 

153 .. admonition:: Scheduler max interval variance 

154 

155 The default max loop interval may vary for different schedulers. 

156 For the default scheduler the value is 5 minutes, but for example 

157 the :pypi:`django-celery-beat` database scheduler the value 

158 is 5 seconds. 

159 """ 

160 last_run_at = self.maybe_make_aware(last_run_at) 

161 rem_delta = self.remaining_estimate(last_run_at) 

162 remaining_s = max(rem_delta.total_seconds(), 0) 

163 if remaining_s == 0: 

164 return schedstate(is_due=True, next=self.seconds) 

165 return schedstate(is_due=False, next=remaining_s)

190class crontab_parser: 

191 """Parser for Crontab expressions. 

192 

193 Any expression of the form 'groups' 

194 (see BNF grammar below) is accepted and expanded to a set of numbers. 

195 These numbers represent the units of time that the Crontab needs to 

196 run on: 

197 

198 .. code-block:: bnf 

199 

200 digit :: '0'..'9' 

201 dow :: 'a'..'z' 

202 number :: digit+ | dow+ 

203 steps :: number 

204 range :: number ( '-' number ) ? 

205 numspec :: '*' | range 

206 expr :: numspec ( '/' steps ) ? 

207 groups :: expr ( ',' expr ) * 

208 

209 The parser is a general purpose one, useful for parsing hours, minutes and 

210 day of week expressions. Example usage: 

211 

212 .. code-block:: pycon 

213 

214 >>> minutes = crontab_parser(60).parse('*/15') 

215 [0, 15, 30, 45] 

216 >>> hours = crontab_parser(24).parse('*/4') 

217 [0, 4, 8, 12, 16, 20] 

218 >>> day_of_week = crontab_parser(7).parse('*') 

219 [0, 1, 2, 3, 4, 5, 6] 

220 

221 It can also parse day of month and month of year expressions if initialized 

222 with a minimum of 1. Example usage: 

223 

224 .. code-block:: pycon 

225 

226 >>> days_of_month = crontab_parser(31, 1).parse('*/3') 

227 [1, 4, 7, 10, 13, 16, 19, 22, 25, 28, 31] 

228 >>> months_of_year = crontab_parser(12, 1).parse('*/2') 

229 [1, 3, 5, 7, 9, 11] 

230 >>> months_of_year = crontab_parser(12, 1).parse('2-12/2') 

231 [2, 4, 6, 8, 10, 12] 

232 

233 The maximum possible expanded value returned is found by the formula: 

234 

235 :math:`max_ + min_ - 1` 

236 """ 

237 

238 ParseException = ParseException 

239 

240 _range = r'(\w+?)-(\w+)' 

241 _steps = r'/(\w+)?' 

242 _star = r'\*' 

243 

244 def __init__(self, max_=60, min_=0): 

245 self.max_ = max_ 

246 self.min_ = min_ 

247 self.pats = ( 

248 (re.compile(self._range + self._steps), self._range_steps), 

249 (re.compile(self._range), self._expand_range), 

250 (re.compile(self._star + self._steps), self._star_steps), 

251 (re.compile('^' + self._star + '$'), self._expand_star), 

252 ) 

253 

254 def parse(self, spec): 

255 acc = set() 

256 for part in spec.split(','): 

257 if not part: 

258 raise self.ParseException('empty part') 

259 acc |= set(self._parse_part(part)) 

260 return acc 

261 

262 def _parse_part(self, part): 

263 for regex, handler in self.pats: 

264 m = regex.match(part) 

265 if m: 

266 return handler(m.groups()) 

267 return self._expand_range((part,)) 

268 

269 def _expand_range(self, toks): 

270 fr = self._expand_number(toks[0]) 

271 if len(toks) > 1: 

272 to = self._expand_number(toks[1]) 

273 if to < fr: # Wrap around max_ if necessary 

274 return (list(range(fr, self.min_ + self.max_)) + 

275 list(range(self.min_, to + 1))) 

276 return list(range(fr, to + 1)) 

277 return [fr] 

278 

279 def _range_steps(self, toks): 

280 if len(toks) != 3 or not toks[2]: 

281 raise self.ParseException('empty filter') 

282 return self._expand_range(toks[:2])[::int(toks[2])] 

283 

284 def _star_steps(self, toks): 

285 if not toks or not toks[0]: 

286 raise self.ParseException('empty filter') 

287 return self._expand_star()[::int(toks[0])] 

288 

289 def _expand_star(self, *args): 

290 return list(range(self.min_, self.max_ + self.min_)) 

291 

292 def _expand_number(self, s): 

293 if isinstance(s, str) and s[0] == '-': 

294 raise self.ParseException('negative numbers not supported') 

295 try: 

296 i = int(s) 

297 except ValueError: 

298 try: 

299 i = weekday(s) 

300 except KeyError: 

301 raise ValueError(f'Invalid weekday literal {s!r}.') 

302 

303 max_val = self.min_ + self.max_ - 1 

304 if i > max_val: 

305 raise ValueError( 

306 f'Invalid end range: {i} > {max_val}.') 

307 if i < self.min_: 

308 raise ValueError( 

309 f'Invalid beginning range: {i} < {self.min_}.') 

310 

311 return i

314class crontab(BaseSchedule): 

315 """Crontab schedule. 

316 

317 A Crontab can be used as the ``run_every`` value of a 

318 periodic task entry to add :manpage:`crontab(5)`-like scheduling. 

319 

320 Like a :manpage:`cron(5)`-job, you can specify units of time of when 

321 you'd like the task to execute. It's a reasonably complete 

322 implementation of :command:`cron`'s features, so it should provide a fair 

323 degree of scheduling needs. 

324 

325 You can specify a minute, an hour, a day of the week, a day of the 

326 month, and/or a month in the year in any of the following formats: 

327 

328 .. attribute:: minute 

329 

330 - A (list of) integers from 0-59 that represent the minutes of 

331 an hour of when execution should occur; or 

332 - A string representing a Crontab pattern. This may get pretty 

333 advanced, like ``minute='*/15'`` (for every quarter) or 

334 ``minute='1,13,30-45,50-59/2'``. 

335 

336 .. attribute:: hour 

337 

338 - A (list of) integers from 0-23 that represent the hours of 

339 a day of when execution should occur; or 

340 - A string representing a Crontab pattern. This may get pretty 

341 advanced, like ``hour='*/3'`` (for every three hours) or 

342 ``hour='0,8-17/2'`` (at midnight, and every two hours during 

343 office hours). 

344 

345 .. attribute:: day_of_week 

346 

347 - A (list of) integers from 0-6, where Sunday = 0 and Saturday = 

348 6, that represent the days of a week that execution should 

349 occur. 

350 - A string representing a Crontab pattern. This may get pretty 

351 advanced, like ``day_of_week='mon-fri'`` (for weekdays only). 

352 (Beware that ``day_of_week='*/2'`` does not literally mean 

353 'every two days', but 'every day that is divisible by two'!) 

354 

355 .. attribute:: day_of_month 

356 

357 - A (list of) integers from 1-31 that represents the days of the 

358 month that execution should occur. 

359 - A string representing a Crontab pattern. This may get pretty 

360 advanced, such as ``day_of_month='2-30/2'`` (for every even 

361 numbered day) or ``day_of_month='1-7,15-21'`` (for the first and 

362 third weeks of the month). 

363 

364 .. attribute:: month_of_year 

365 

366 - A (list of) integers from 1-12 that represents the months of 

367 the year during which execution can occur. 

368 - A string representing a Crontab pattern. This may get pretty 

369 advanced, such as ``month_of_year='*/3'`` (for the first month 

370 of every quarter) or ``month_of_year='2-12/2'`` (for every even 

371 numbered month). 

372 

373 .. attribute:: nowfun 

374 

375 Function returning the current date and time 

376 (:class:`~datetime.datetime`). 

377 

378 .. attribute:: app 

379 

380 The Celery app instance. 

381 

382 It's important to realize that any day on which execution should 

383 occur must be represented by entries in all three of the day and 

384 month attributes. For example, if ``day_of_week`` is 0 and 

385 ``day_of_month`` is every seventh day, only months that begin 

386 on Sunday and are also in the ``month_of_year`` attribute will have 

387 execution events. Or, ``day_of_week`` is 1 and ``day_of_month`` 

388 is '1-7,15-21' means every first and third Monday of every month 

389 present in ``month_of_year``. 

390 """ 

391 

392 def __init__(self, minute='*', hour='*', day_of_week='*',

399 self._orig_kwargs = kwargs

402 self.day_of_week = self._expand_cronspec(day_of_week, 7) 

403 self.day_of_month = self._expand_cronspec(day_of_month, 31, 1) 

404 self.month_of_year = self._expand_cronspec(month_of_year, 12, 1)

408 def _expand_cronspec(cronspec, max_, min_=0): 

409 """Expand cron specification. 

410 

411 Takes the given cronspec argument in one of the forms: 

412 

413 .. code-block:: text 

414 

415 int (like 7) 

416 str (like '3-5,*/15', '*', or 'monday') 

417 set (like {0,15,30,45} 

418 list (like [8-17]) 

419 

420 And convert it to an (expanded) set representing all time unit 

421 values on which the Crontab triggers. Only in case of the base 

422 type being :class:`str`, parsing occurs. (It's fast and 

423 happens only once for each Crontab instance, so there's no 

424 significant performance overhead involved.) 

425 

426 For the other base types, merely Python type conversions happen. 

427 

428 The argument ``max_`` is needed to determine the expansion of 

429 ``*`` and ranges. The argument ``min_`` is needed to determine 

430 the expansion of ``*`` and ranges for 1-based cronspecs, such as 

431 day of month or month of year. The default is sufficient for minute, 

432 hour, and day of week. 

433 """ 

434 if isinstance(cronspec, numbers.Integral): 

435 result = {cronspec} 

436 elif isinstance(cronspec, str): 

437 result = crontab_parser(max_, min_).parse(cronspec) 

438 elif isinstance(cronspec, set): 

439 result = cronspec 

440 elif isinstance(cronspec, Iterable): 

441 result = set(cronspec) 

442 else: 

443 raise TypeError(CRON_INVALID_TYPE.format(type=type(cronspec))) 

444 

445 # assure the result does not preceed the min or exceed the max 

446 for number in result: 

447 if number >= max_ + min_ or number < min_: 

448 raise ValueError(CRON_PATTERN_INVALID.format( 

449 min=min_, max=max_ - 1 + min_, value=number)) 

450 return result

452 def _delta_to_next(self, last_run_at, next_hour, next_minute): 

453 """Find next delta. 

454 

455 Takes a :class:`~datetime.datetime` of last run, next minute and hour, 

456 and returns a :class:`~celery.utils.time.ffwd` for the next 

457 scheduled day and time. 

458 

459 Only called when ``day_of_month`` and/or ``month_of_year`` 

460 cronspec is specified to further limit scheduled task execution. 

461 """ 

462 datedata = AttributeDict(year=last_run_at.year) 

463 days_of_month = sorted(self.day_of_month) 

464 months_of_year = sorted(self.month_of_year) 

465 

466 def day_out_of_range(year, month, day): 

467 try: 

468 datetime(year=year, month=month, day=day) 

469 except ValueError: 

470 return True 

471 return False 

472 

473 def is_before_last_run(year, month, day): 

474 return self.maybe_make_aware(datetime(year, 

475 month, 

476 day)) < last_run_at 

477 

478 def roll_over(): 

479 for _ in range(2000): 

480 flag = (datedata.dom == len(days_of_month) or 

481 day_out_of_range(datedata.year, 

482 months_of_year[datedata.moy], 

483 days_of_month[datedata.dom]) or 

484 (is_before_last_run(datedata.year, 

485 months_of_year[datedata.moy], 

486 days_of_month[datedata.dom]))) 

487 

488 if flag: 

489 datedata.dom = 0 

490 datedata.moy += 1 

491 if datedata.moy == len(months_of_year): 

492 datedata.moy = 0 

493 datedata.year += 1 

494 else: 

495 break 

496 else: 

497 # Tried 2000 times, we're most likely in an infinite loop 

498 raise RuntimeError('unable to rollover, ' 

499 'time specification is probably invalid') 

500 

501 if last_run_at.month in self.month_of_year: 

502 datedata.dom = bisect(days_of_month, last_run_at.day) 

503 datedata.moy = bisect_left(months_of_year, last_run_at.month) 

504 else: 

505 datedata.dom = 0 

506 datedata.moy = bisect(months_of_year, last_run_at.month) 

507 if datedata.moy == len(months_of_year): 

508 datedata.moy = 0 

509 roll_over() 

510 

511 while 1: 

512 th = datetime(year=datedata.year, 

513 month=months_of_year[datedata.moy], 

514 day=days_of_month[datedata.dom]) 

515 if th.isoweekday() % 7 in self.day_of_week: 

516 break 

517 datedata.dom += 1 

518 roll_over() 

519 

520 return ffwd(year=datedata.year, 

521 month=months_of_year[datedata.moy], 

522 day=days_of_month[datedata.dom], 

523 hour=next_hour, 

524 minute=next_minute, 

525 second=0, 

526 microsecond=0)

543 def remaining_delta(self, last_run_at, tz=None, ffwd=ffwd): 

544 # pylint: disable=redefined-outer-name 

545 # caching global ffwd 

546 tz = tz or self.tz 

547 last_run_at = self.maybe_make_aware(last_run_at) 

548 now = self.maybe_make_aware(self.now()) 

549 dow_num = last_run_at.isoweekday() % 7 # Sunday is day 0, not day 7 

550 

551 execute_this_date = ( 

552 last_run_at.month in self.month_of_year and 

553 last_run_at.day in self.day_of_month and 

554 dow_num in self.day_of_week 

555 ) 

556 

557 execute_this_hour = ( 

558 execute_this_date and 

559 last_run_at.day == now.day and 

560 last_run_at.month == now.month and 

561 last_run_at.year == now.year and 

562 last_run_at.hour in self.hour and 

563 last_run_at.minute < max(self.minute) 

564 ) 

565 

566 if execute_this_hour: 

567 next_minute = min(minute for minute in self.minute 

568 if minute > last_run_at.minute) 

569 delta = ffwd(minute=next_minute, second=0, microsecond=0) 

570 else: 

571 next_minute = min(self.minute) 

572 execute_today = (execute_this_date and 

573 last_run_at.hour < max(self.hour)) 

574 

575 if execute_today: 

576 next_hour = min(hour for hour in self.hour 

577 if hour > last_run_at.hour) 

578 delta = ffwd(hour=next_hour, minute=next_minute, 

579 second=0, microsecond=0) 

580 else: 

581 next_hour = min(self.hour) 

582 all_dom_moy = (self._orig_day_of_month == '*' and 

583 self._orig_month_of_year == '*') 

584 if all_dom_moy: 

585 next_day = min([day for day in self.day_of_week 

586 if day > dow_num] or self.day_of_week) 

587 add_week = next_day == dow_num 

588 

589 delta = ffwd( 

590 weeks=add_week and 1 or 0, 

591 weekday=(next_day - 1) % 7, 

592 hour=next_hour, 

593 minute=next_minute, 

594 second=0, 

595 microsecond=0, 

596 ) 

597 else: 

598 delta = self._delta_to_next(last_run_at, 

599 next_hour, next_minute) 

600 return self.to_local(last_run_at), delta, self.to_local(now)

770 def remaining_estimate(self, last_run_at): 

771 """Return estimate of next time to run. 

772 

773 Returns: 

774 ~datetime.timedelta: when the periodic task should 

775 run next, or if it shouldn't run today (e.g., the sun does 

776 not rise today), returns the time when the next check 

777 should take place. 

778 """ 

779 last_run_at = self.maybe_make_aware(last_run_at) 

780 last_run_at_utc = localize(last_run_at, timezone.utc) 

781 self.cal.date = last_run_at_utc 

782 try: 

783 if self.use_center: 

784 next_utc = getattr(self.cal, self.method)( 

785 self.ephem.Sun(), 

786 start=last_run_at_utc, use_center=self.use_center 

787 ) 

788 else: 

789 next_utc = getattr(self.cal, self.method)( 

790 self.ephem.Sun(), start=last_run_at_utc 

791 ) 

792 

793 except self.ephem.CircumpolarError: # pragma: no cover 

794 # Sun won't rise/set today. Check again tomorrow 

795 # (specifically, after the next anti-transit). 

796 next_utc = ( 

797 self.cal.next_antitransit(self.ephem.Sun()) + 

798 timedelta(minutes=1) 

799 ) 

800 next = self.maybe_make_aware(next_utc.datetime()) 

801 now = self.maybe_make_aware(self.now()) 

802 delta = next - now 

803 return delta

805 def is_due(self, last_run_at): 

806 """Return tuple of ``(is_due, next_time_to_run)``. 

807 

808 Note: 

809 next time to run is in seconds. 

810 

811 See Also: 

812 :meth:`celery.schedules.schedule.is_due` for more information. 

813 """ 

814 rem_delta = self.remaining_estimate(last_run_at) 

815 rem = max(rem_delta.total_seconds(), 0) 

816 due = rem == 0 

817 if due: 

818 rem_delta = self.remaining_estimate(self.now()) 

819 rem = max(rem_delta.total_seconds(), 0) 

820 return schedstate(due, rem)
