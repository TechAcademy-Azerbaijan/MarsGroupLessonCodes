create table phone_numbers(
   id int(11) unsigned AUTO_INCREMENT PRIMARY KEY,
   number varchar(15) NOT NULL,
   user_id int(11) NOT NULL
   foreign key user_id REFERENCES users(id) ON DELETE CASCADE
);
